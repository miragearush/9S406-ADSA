/**
* 994. Rotting Oranges
* Link : https://leetcode.com/problems/rotting-oranges/description/
* Topic : Graph
*/
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m=grid.size();
        int n=grid[0].size();
        vector<vector<int>>visited(m,vector<int>(n,0));
        queue<pair<int,int>>q;
        //all rotten or starting point are in the queue
        int fresh=0;
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j]==2)
                {
                    q.push({i,j});
                    visited[i][j]=1;
                }
                if(grid[i][j]==1)
                    fresh++;
            }
        }
        
        int ans=0;
        int dx[]={0,1,0,-1};
        int dy[]={1,0,-1,0};
        while(!q.empty())
        {
            int sz=q.size();
            for(int i=0;i<sz;i++)
            {
                int x=q.front().first;
                int y=q.front().second;
                q.pop();
                
                for(int i=0;i<4;i++) {
                    int newX=x+dx[i];
                    int newY=y+dy[i];

                    if(newX>=0 && newX<m && newY>=0 && newY<n && grid[newX][newY]==1 && visited[newX][newY]==0) {
                        q.push({newX,newY});
                        visited[newX][newY]=1;
                        fresh--;
                    }
                }
            }
            if(!q.empty())
                ans++;
        }
        if(fresh!=0)
            return -1;
        return ans;
    }
};
