/**
* Number of Connected Components
* Link : https://www.geeksforgeeks.org/problems/number-of-provinces/1
* Topic : Graph
*/
class Solution {
  public:
    void dfs(int node,vector<vector<int>>&adj,vector<int>&visited) {
        visited[node]=1;
        
        //explore all its children nodes
        for(int v:adj[node]) {
            if(!visited[v]) {
                dfs(v,adj,visited);
            }
        }
    }
    int countConnected(int V, vector<vector<int>>& edges) {
        // code here
        vector<vector<int>>adj(V);
        //to adjajency list
        for(int i=0;i<edges.size();i++) {
            int u=edges[i][0];
            int v=edges[i][1];
            
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
        vector<int>visited(V,0);
        int count=0;
        
        for(int i=0;i<V;i++) {
            if(!visited[i]) {
                dfs(i,adj,visited);
                count++;
            }
        }
        return count;
    }
};
