/**
* Maximum Meetings in One Room
* Link : https://www.geeksforgeeks.org/problems/maximum-meetings-in-one-room/1
* Topic : Greedy
*/
class Solution {
  public:
    vector<int> maxMeetings(vector<int> &s, vector<int> &f) {
        // code here
        int n=s.size();
        vector<pair<int,pair<int,int>>>vp;
        for(int i=0;i<n;i++) {
            vp.push_back({f[i],{i,s[i]}});
        }
        
        sort(vp.begin(),vp.end());
        
        int lastEnded=-1;
        vector<int>ans;
        
        for(int i=0;i<n;i++) {
            int start=vp[i].second.second;
            int end=vp[i].first;
            int index=vp[i].second.first;
            
            if(lastEnded<start) {
                ans.push_back(index+1);
                lastEnded=end;
            }
        }
        sort(ans.begin(),ans.end());
        return ans;
    }
};
