/**
* DFS of Graph
* Link : https://www.geeksforgeeks.org/problems/depth-first-traversal-for-a-graph/1
* Topic : Graph
*/
class Solution {
  public:
    void findDfs(int node,vector<vector<int>>&adj,vector<int>&visited
    ,vector<int>&ans) {
        //mark the node as visited
        visited[node]=1;
        ans.push_back(node);
        
        //explore all its child
        for(int v:adj[node]) {
            if(!visited[v]) {
                findDfs(v,adj,visited,ans);
            }
        }
    }
    vector<int> dfs(vector<vector<int>>& adj) {
        // Code here
        int n=adj.size();
        vector<int>visited(n,0);
        vector<int>ans;
        findDfs(0,adj,visited,ans);
        return ans;
    }
};
