class Solution {
  public:
    Node* LCA(Node* root, Node* n1, Node* n2) {
        // code here
        Node *curr=root;
        while(curr) {
            //both smaller
            if(n1->data<curr->data && n2->data<curr->data) {
                curr=curr->left;
            }
            else if(n1->data>curr->data && n2->data>curr->data) {
                curr=curr->right;
            }
            else 
                return curr;
        }
        return NULL;
    }
};
