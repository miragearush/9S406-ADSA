class Solution {
  public:
    bool isValid(Node *node,int mini,int maxi) {
        if(node==NULL)
            return true;
        //check value
        if(node->data<=mini || node->data>=maxi)
            return false;
        bool left=isValid(node->left,mini,node->data);
        bool right=isValid(node->right,node->data,maxi);
        
        return left && right;
    }
    bool isBST(Node* root) {
        // code h
        return isValid(root,INT_MIN,INT_MAX);
        
    }
};
