/**
* BFS of graph
* Link : https://www.geeksforgeeks.org/problems/bfs-traversal-of-graph/1
* Topic : Graph
*/
class Solution {
  public:
    vector<int> bfs(vector<vector<int>> &adj) {
        // code here
        int n=adj.size();
        queue<int>q;
        vector<int>visited(n,0);
        
        q.push(0);
        visited[0]=1;
        
        vector<int>ans;
        
        while(!q.empty()) {
            int u=q.front();
            ans.push_back(u);
            q.pop();
            
            for(int v:adj[u]) {
                if(!visited[v]) {
                    q.push(v);
                    visited[v]=1;
                }
            }
        }
        return ans;
    }
};
