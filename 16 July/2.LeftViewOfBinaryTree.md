class Solution {
  public:
    void preorder(Node *node,int level,vector<int>&ans) {
        if(node==NULL)
            return;
        
        //if size == level -> push
        if(level==ans.size())
            ans.push_back(node->data);
        //visit left child
        preorder(node->left,level+1,ans);
        //visit right child
        preorder(node->right,level+1,ans);
        
    }
    vector<int> leftView(Node *root) {
        // code here
        vector<int>ans;
        preorder(root,0,ans);
        return ans;
    }
};
