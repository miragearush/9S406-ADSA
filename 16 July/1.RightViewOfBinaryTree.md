/**
* Right View Of Binary Tree
* Link : https://www.geeksforgeeks.org/problems/right-view-of-binary-tree/1
* Topic : Trees
*/
/*
Definition for Node
class Node {
  public:
    int data;
    Node* left;
    Node* right;

    Node(int val) {
        data = val;
        left = right = nullptr;
    }
};
*/

class Solution {
  public:
    void reversePreorder(Node *node,int level,vector<int>&ans) {
        if(node==NULL)
            return;
        
        //if size == level -> push
        if(level==ans.size())
            ans.push_back(node->data);
        //visit right child
        reversePreorder(node->right,level+1,ans);
        //visit left child
        reversePreorder(node->left,level+1,ans);
    }
    vector<int> rightView(Node *root) {
        //  code here
        vector<int>ans;
        reversePreorder(root,0,ans);
        return ans;
    }
};
