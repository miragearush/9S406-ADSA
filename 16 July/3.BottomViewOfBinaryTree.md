/**
* Bottom View of Binary Tree
* Link : https://www.geeksforgeeks.org/problems/bottom-view-of-binary-tree/1
* Topic : Trees
*/
/*
Definition for Node
class Node {
public:
    int data;
    Node* left;
    Node* right;

    Node(int val) {
        data = val;
        left = right = nullptr;
    }
};
*/

class Solution {
  public:
    vector<int> bottomView(Node *root) {
        // code here
        map<int,int>mpp;
        queue<pair<Node*,int>>q;
        //initial setup
        q.push({root,0});
        
        while(!q.empty()) {
            Node *curr=q.front().first;
            int level=q.front().second;
            q.pop();
            //push it into map
            mpp[level]=curr->data;
            //check left child
            if(curr->left)
                q.push({curr->left,level-1});
            if(curr->right)
                q.push({curr->right,level+1});
        }
        
        vector<int>ans;
        for(auto it:mpp) {
            ans.push_back(it.second);
        }
        return ans;
        
    }
};
