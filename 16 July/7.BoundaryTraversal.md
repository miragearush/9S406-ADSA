/**
* Tree Boundary Traversal
* Link : https://www.geeksforgeeks.org/problems/boundary-traversal-of-binary-tree/1
* Topic : Trees
*/
/* Node Structure
class Node {
  public:
    int data;
    Node* left, *right;
    Node(int val) {
        data = val;
        left = right = nullptr;
    }
}; */

class Solution {
  public:
    void inorder(Node *node,vector<int>&ans) {
        if(node==NULL)
            return;
        inorder(node->left,ans);
        if(node->left==NULL && node->right==NULL)
            ans.push_back(node->data);
        inorder(node->right,ans);
    }
    vector<int> boundaryTraversal(Node *root) {
        // code here
        vector<int>ans;
        //root inserted
        if(!(root->left==NULL && root->right==NULL)) {
            ans.push_back(root->data);
        }
        //insert left elements
        Node *temp=root->left;
        while(temp!=NULL) {
            if(!(temp->left==NULL && temp->right==NULL))
                ans.push_back(temp->data);
            if(temp->left)
                temp=temp->left;
            else
                temp=temp->right;
        }
        //insert leaf elements
        inorder(root,ans);
        //insert right elements
        temp=root->right;
        stack<int>st;
        while(temp!=NULL) {
            if(!(temp->left==NULL && temp->right==NULL))
                st.push(temp->data);
            if(temp->right)
                temp=temp->right;
            else
                temp=temp->left;
        }
        while(!st.empty()) {
            ans.push_back(st.top());
            st.pop();
        }
        return ans;
    }
};
