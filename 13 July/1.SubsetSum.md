/**
* Subset Sum Problem
* Link : https://www.geeksforgeeks.org/problems/subset-sum-problem-1611555638/1
* Topic : Recursion 
*/
class Solution {
  public:
    bool solve(int idx,int currSum,int target,vector<int>&arr) {
        //base case
        if(currSum==target)
            return true;
        if(idx==arr.size() || currSum>target) {
            return false;
        }
        
        //option 1 : 
        bool pick = solve(idx+1,currSum+arr[idx],target,arr);
        if(pick==true)
            return true;
        bool notpick = solve(idx+1,currSum,target,arr);
        return notpick;
    }
    bool isSubsetSum(vector<int>& arr, int sum) {
        // code here
        return solve(0,0,sum,arr);
    }
};
