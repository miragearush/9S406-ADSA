/**
* 216. Combination Sum III
* Link : https://leetcode.com/problems/combination-sum-iii/
* Topic : Recursion & Backtracking
*/
class Solution {
public:
    void solve(int idx,int k,int currSum,int target,vector<int>&curr,vector<vector<int>>&ans) {
        //base case
        if(curr.size()==k) {
            if(currSum==target) {
                ans.push_back(curr);
            }
            return;
        }
        //try out all possible combinations
        for(int i=idx;i<=9;i++) {
            //no need of adding if sum exceeds
            if(currSum+i>target)
                break;
            curr.push_back(i);
            solve(i+1,k,currSum+i,target,curr,ans);
            curr.pop_back();
        }
    }
    vector<vector<int>> combinationSum3(int k, int n) {
        vector<int>curr;
        vector<vector<int>>ans;
        solve(1,k,0,n,curr,ans);
        return ans;
    }
};
