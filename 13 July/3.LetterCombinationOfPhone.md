/**
* 17. Letter Combinations of a Phone Number
* Link : https://leetcode.com/problems/letter-combinations-of-a-phone-number/description/
* Topic : Recursion & Backtracking
*/
class Solution {
public:
    void solve(int idx,string curr,vector<string>&ans,string digits,vector<string>&map) {
        //base case
        if(idx==digits.size()) {
            ans.push_back(curr);
            return;
        }
        //explore all the options
        string s=map[digits[idx]-'0'];
        //iterate over the string and try all combinations
        for(char ch:s) {
            curr.push_back(ch);
            solve(idx+1,curr,ans,digits,map);
            curr.pop_back();
        }
    }
    vector<string> letterCombinations(string digits) {
        vector<string>map={"","","abc","def","ghi","jkl",
                            "mno","pqrs","tuv","wxyz"}; 
        vector<string>ans;
        solve(0,"",ans,digits,map);
        return ans;
    }
};
