/**
*  Longest Substring with At Most K Distinct Characters
* Link : https://www.naukri.com/code360/problems/distinct-characters_2221410
* Topic : Two Pointers
*/
#include<bits/stdc++.h>
int kDistinctChars(int k, string &s)
{
    // Write your code here
    map<char,int>mpp;
    int l=0;
    int ans=0;
    for(int r=0;r<s.size();r++) {
        mpp[s[r]]++;

        //delete
        while(mpp.size()>k) {
            mpp[s[l]]--;

            if(mpp[s[l]]==0) {
                mpp.erase(mpp.find(s[l]));
            }
            l++;
        }

        //update ans
        ans=max(ans,r-l+1);
    }
    return ans;
}
