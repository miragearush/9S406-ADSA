/**
* 992. Subarrays with K Different Integers
* Link : https://leetcode.com/problems/subarrays-with-k-different-integers/description/
* Topic : Two Pointers
*/
class Solution {
public:
    int countSubarray(vector<int>&s,int k) {
        map<int,int>mpp;
        int l=0;
        int ans=0;
        for(int r=0;r<s.size();r++) {
            mpp[s[r]]++;

            //delete
            while(mpp.size()>k) {
                mpp[s[l]]--;

                if(mpp[s[l]]==0) {
                    mpp.erase(mpp.find(s[l]));
                }
                l++;
            }

            //update ans
            ans+=(r-l+1);
        }
        return ans;
    }
    int subarraysWithKDistinct(vector<int>& nums, int k) {
        return countSubarray(nums,k)-countSubarray(nums,k-1);
    }
};
