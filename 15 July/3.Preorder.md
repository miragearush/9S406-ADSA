class Solution {
  public:
  void preorder(Node *node,vector<int>&ans) {
        //base case
        if(node==NULL)
            return;
        ans.push_back(node->data);
        preorder(node->left,ans);
        preorder(node->right,ans);
        
    }
    vector<int> preOrder(Node* root) {
        // code here
        vector<int>ans;
        preorder(root,ans);
        return ans;
    }
};
