/**
* Level Order Traversal
* Link : https://www.geeksforgeeks.org/problems/level-order-traversal/1
* Topic : Trees

/* A binary tree Node
class Node {
  public:
    int data;
    Node* left;
    Node* right;

    // Constructor
    Node(int val) {
        data = val;
        left = nullptr;
        right = nullptr;
    }
};
*/

class Solution {
  public:
    vector<int> levelOrder(Node *root) {
        // code here
        queue<Node *>q;
        //push root in the queue
        q.push(root);
        
        vector<int>ans;
        //start the iteration
        while(!q.empty()) {
            Node *curr=q.front();
            q.pop();
            
            //store in the ans
            ans.push_back(curr->data);
            //check if left child exists
            if(curr->left)
                q.push(curr->left);
            //check if right child exists
            if(curr->right)
                q.push(curr->right);
        }
        return ans;
    }
};
