class Solution {
  public:
    void inorder(Node *node,vector<int>&ans) {
        if(node==NULL)
            return;
        inorder(node->left,ans);
        ans.push_back(node->data);
        inorder(node->right,ans);
    }
    vector<int> inOrder(Node* root) {
        // code here
        vector<int>ans;
        inorder(root,ans);
        return ans;
        
    }
