class Solution {
  public:
    void postorder(Node *node,vector<int>&ans) {
        //base case
        if(node==NULL)
            return;
        postorder(node->left,ans);
        postorder(node->right,ans);
        ans.push_back(node->data);
    }
    vector<int> postOrder(Node* root) {
        // code here
        vector<int>ans;
        postorder(root,ans);
        return ans;
        
    }
};
