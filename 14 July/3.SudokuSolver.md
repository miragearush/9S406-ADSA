/**
* 37. Sudoku Solver
* Link : https://leetcode.com/problems/sudoku-solver/
* Topic : Recursion & Backtracking
*/
class Solution {
public:
    bool isPossible(int row,int col,char x,vector<vector<char>>&board) {
        for(int i=0;i<9;i++) {
            //check row
            if(board[row][i]==x)
                return false;
            //check col
            if(board[i][col]==x)
                return false;
            //check the box
            int r=((row/3)*3)+(i/3);
            int c=((col/3)*3)+(i%3);
            
            if(board[r][c]==x)
                return false;
        }
        return true;
    }
    bool solve(vector<vector<char>>&board) {
        for(int i=0;i<9;i++) {
            for(int j=0;j<9;j++) {
                // look for answer only if there is dot
                if(board[i][j]=='.') {
                    for(char x='1';x<='9';x++) {
                        if(isPossible(i,j,x,board)) {
                            board[i][j]=x;
                            if(solve(board)==true)
                                return true;
                            board[i][j]='.';
                        }
                    }
                    return false;
                }
            }
        }
        return true;
    }
    void solveSudoku(vector<vector<char>>& board) {
        solve(board);
    }
};
