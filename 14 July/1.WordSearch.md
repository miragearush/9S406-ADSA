/**
* 79. Word Search
* Link : https://leetcode.com/problems/word-search/description/
* Topic : Recursion & Backtracking
*/

class Solution {
public:
    bool solve(int row,int col,int idx,string word,vector<vector<char>>&board,vector<vector<int>>&visited) {
        //base case
        if(idx==word.size()) {
            return true;
        }
        int dx[]={0,1,0,-1};
        int dy[]={1,0,-1,0};
        int m=board.size();
        int n=board[0].size();

        for(int i=0;i<4;i++) {
            int newX=row+dx[i];
            int newY=col+dy[i];
            
            //can we go to a cell or not
            if(newX>=0 && newX<m && newY>=0 && newY<n && visited[newX][newY]==0 
                && word[idx]==board[newX][newY]) {
                    visited[newX][newY]=1;
                    if(solve(newX,newY,idx+1,word,board,visited)==true)
                        return true;
                    visited[newX][newY]=0;
                }
        }
        return false;
    }
    bool exist(vector<vector<char>>& board, string word) {
        int m=board.size();
        int n=board[0].size();
        
        vector<vector<int>>visited(m,vector<int>(n,0));

        for(int i=0;i<m;i++) {
            for(int j=0;j<n;j++) {
                if(board[i][j]==word[0]) {
                    visited[i][j]=1;
                    if(solve(i,j,1,word,board,visited)==true)
                        return true;
                    visited[i][j]=0;
                }
            }
        }
        return false;
    }
};
