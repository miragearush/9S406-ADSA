/**
* Rat in a Maze
* Link : https://www.geeksforgeeks.org/problems/rat-in-a-maze-problem/1
* Topic : Recursion & Backtracking
*/
class Solution {
  public:
    void solve(int row,int col,string path,vector<string>&ans,vector<vector<int>>&maze,vector<vector<int>>&visited) {
        //base case
        int n=maze.size();
        //base case
        if(row==n-1 && col==n-1) {
            ans.push_back(path);
            return;
        }
        int dx[]={1,0,0,-1};
        int dy[]={0,-1,1,0};
        int dir[]={'D','L','R','U'};

        for(int i=0;i<4;i++) {
            int newX=row+dx[i];
            int newY=col+dy[i];
            
            //can we go to a cell or not
            if(newX>=0 && newX<n && newY>=0 && newY<n && visited[newX][newY]==0 
                && maze[newX][newY]==1) {
                    visited[newX][newY]=1;
                    path.push_back(dir[i]);
                    solve(newX,newY,path,ans,maze,visited);
                    path.pop_back();
                    visited[newX][newY]=0;
                }
        }
    }
    vector<string> ratInMaze(vector<vector<int>>& maze) {
        // code here
        int n=maze.size();
        vector<string>ans;
        if(maze[0][0]==0)
            return ans;
            
        vector<vector<int>>visited(n,vector<int>(n,0));
        visited[0][0]=1;
        solve(0,0,"",ans,maze,visited);
        return ans;
    }
};
