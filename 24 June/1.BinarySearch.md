/**
 * Binary Search Implementation
 * Link : https://www.geeksforgeeks.org/problems/who-will-win-1587115621/1
 * Topic : Binary Search
 */
class Solution {
  public:
    bool binarySearch(vector<int>& arr, int k) {
        int low=0,high=arr.size()-1;
        
        while(low<=high) {
            int mid=low+(high-low)/2;
            
            if(arr[mid]==k)
                return true;
                
            if(arr[mid]<k)
                low=mid+1;
            else
                high=mid-1;
        }
        return false;
        
    }
};
