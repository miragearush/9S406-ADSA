/**
 * Factory Machines
 * Link : https://cses.fi/problemset/task/1620/
 * Topic : Binary Search on Answers
 */
#include<bits/stdc++.h>
using namespace std;
 
typedef long long ll;
 
bool possible(ll mid,ll a[],ll n,ll t)
{
    ll done=0;
    for(int i=0;i<n;i++)
    {
        done+=((ll)mid/a[i]);
        if(done>=t)
            return true;
    }
    return done>=t;   
}
 
int main(){
    ll n,t;
    cin>>n>>t;
    ll a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    ll low=0,high=1e18,ans=-1;
    while(low<=high)
    {
        ll mid=low+((high-low)/2);
        if(possible(mid,a,n,t))
        {
            ans=mid;
            high=mid-1;
        }
        else
            low=mid+1;
    }
    cout<<ans;
    return 0;
