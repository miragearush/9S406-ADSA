/**
* Towers 
* Link : https://cses.fi/problemset/task/1073
* Topic : Greedy
*/
#include <bits/stdc++.h>
using namespace std;

int upperBound(vector<int>&arr,int x) {
    int low=0,high=arr.size()-1;
    int ans=-1;

    while(low<=high) {
        int mid=(low+high)/2;
        if(arr[mid]>x) {
            ans=mid;
            high=mid-1;
        }
        else {
            low=mid+1;
        }
    }
    return ans;
}

int main() 
{
    int n;
    cin>>n;
    vector<int>a(n);
    for(int i=0;i<n;i++) {
        cin>>a[i];
    }
    vector<int>towers;
    for(int i=0;i<n;i++) {
        if(towers.size()==0) {
            towers.push_back(a[i]);
        }
        else {
            int idx=upperBound(towers,a[i]);
            if(idx==-1) {
                towers.push_back(a[i]);
            }
            else {
                towers[idx]=a[i];
            }
        }
    }
    cout<<towers.size();
    return 0;
}
