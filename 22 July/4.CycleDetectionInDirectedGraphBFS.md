/**
* Directed Graph Cycle in BFS (Kahn's Algorithm)
* Link : https://www.geeksforgeeks.org/problems/detect-cycle-in-a-directed-graph/1#
* Topic : Graph
*/
class Solution {
  public:
    
    
    bool isCyclic(int V, vector<vector<int>> &edges) {
        // code here
        vector<vector<int>>adj(V);
        vector<int>indegree(V,0);
        for(auto it:edges) {
            int u=it[0];
            int v=it[1];
            adj[u].push_back(v);
            //u->v
            indegree[v]++;
        }
        queue<int>q;
        for(int i=0;i<V;i++) {
            if(indegree[i]==0) {
                q.push(i);
            }
        }
        int ans=0;
        while(!q.empty()) {
            int node=q.front();
            q.pop();
            ans++;
            
            for(int child:adj[node]) {
                indegree[child]--;
                if(indegree[child]==0)
                    q.push(child);
            }
        }
        if(ans==V) {
            return false;
        }
        return true;
        
    }
};
