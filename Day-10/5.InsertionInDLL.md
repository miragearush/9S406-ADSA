class Solution {
  public:
    Node *insertAtPos(Node *head, int p, int x) {

        Node* curr = head;
        // Reach the pth node
        for (int i = 0; i < p; i++) {
            curr = curr->next;
        }
        Node* newNode = new Node(x);
        newNode->next = curr->next;
        newNode->prev = curr;

        if (curr->next != NULL)
            curr->next->prev = newNode;

        curr->next = newNode;

        return head;
    }
};
