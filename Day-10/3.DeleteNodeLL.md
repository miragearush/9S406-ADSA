class Solution {
  public:
    Node* deleteNode(Node* head, int x) {
        // Delete the first node
        if (x == 1) {
            Node* temp = head;
            head = head->next;
            delete temp;
            return head;
        }
        Node* curr = head;

        // Move to the (x-1)th node
        for (int i = 1; i < x - 1; i++) {
            curr = curr->next;
        }
        Node* temp = curr->next;
        curr->next = temp->next;
        delete temp;

        return head;
    }
};
