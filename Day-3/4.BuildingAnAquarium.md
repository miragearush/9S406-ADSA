/**
 * E.Building An Aquarium
 * Link : https://codeforces.com/contest/1873/problem/E
 * Topic : Binary Search on Answers
 */
#include <bits/stdc++.h>
using namespace std;
 
bool check(long long height,vector<int>&a,int x) {
    long long waterReq=0;
    
    for(int i=0;i<a.size();i++) {
        if(height>a[i]) {
            waterReq+=(height-a[i]);
            
            if(waterReq>x)
                return false;
        }
    }
    return true;
}
 
int main() 
{
    int t;
    cin>>t;
    while(t--) {
        int n,x;
        cin>>n>>x;
        
        vector<int>a(n);
        for(int i=0;i<n;i++) {
            cin>>a[i];
        }
        
        long long low=1;
        long long high=(*max_element(a.begin(),a.end())+x);
        
        long long ans=1;
        
        while(low<=high) {
            long long mid=low+(high-low)/2;
            
            if(check(mid,a,x)) {
                ans=mid;
                low=mid+1;
            }
            else {
                high=mid-1;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
