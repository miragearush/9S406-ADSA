/**
* 84. Largest Rectangle in Histogram
* Link : https://leetcode.com/problems/largest-rectangle-in-histogram/description/
* Topic : Stack & Queue
*/
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n=heights.size();

        vector<int>nse(n),pse(n);
        stack<int>st;

        //creating nse
        for(int i=n-1;i>=0;i--) {
            while(!st.empty() && heights[i]<=heights[st.top()]) {
                st.pop();
            }

            if(st.empty()) {
                nse[i]=n;
            }
            else {
                nse[i]=st.top();
            }
            st.push(i);
        }
        while(!st.empty()) {
            st.pop();
        }
        //creating pse
        for(int i=0;i<n;i++) {
            while(!st.empty() && heights[i]<=heights[st.top()]) {
                st.pop();
            }

            if(st.empty()) {
                pse[i]=-1;
            }
            else {
                pse[i]=st.top();
            }
            st.push(i);
        }
        
        int ans=0;
        for(int i=0;i<n;i++) {
            int area=(nse[i]-pse[i]-1)*heights[i];
            ans=max(ans,area);
        }
        return ans;
    }
};
