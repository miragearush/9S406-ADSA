/** 
* 85. Maximal Rectangle
* Link : https://leetcode.com/problems/maximal-rectangle/description/
* Topic : Stack & Queue
*/
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n=heights.size();

        vector<int>nse(n),pse(n);
        stack<int>st;

        //creating nse
        for(int i=n-1;i>=0;i--) {
            while(!st.empty() && heights[i]<=heights[st.top()]) {
                st.pop();
            }

            if(st.empty()) {
                nse[i]=n;
            }
            else {
                nse[i]=st.top();
            }
            st.push(i);
        }
        while(!st.empty()) {
            st.pop();
        }
        //creating pse
        for(int i=0;i<n;i++) {
            while(!st.empty() && heights[i]<=heights[st.top()]) {
                st.pop();
            }

            if(st.empty()) {
                pse[i]=-1;
            }
            else {
                pse[i]=st.top();
            }
            st.push(i);
        }
        
        int ans=0;
        for(int i=0;i<n;i++) {
            int area=(nse[i]-pse[i]-1)*heights[i];
            ans=max(ans,area);
        }
        return ans;
    }
    int maximalRectangle(vector<vector<char>>& matrix) {
        int n=matrix.size();
        int m=matrix[0].size();

        vector<vector<int>>prefix(n,vector<int>(m,0));

        for(int i=0;i<n;i++) {
            for(int j=0;j<m;j++) {
                if(i==0) {
                    prefix[i][j]=matrix[i][j]-'0';
                }
                else {
                    if(matrix[i][j]=='0')
                        prefix[i][j]=0;
                    else
                        prefix[i][j]=prefix[i-1][j]+1;
                }
            }
        }
        int ans=0;
        for(int i=0;i<n;i++) {
            int area=largestRectangleArea(prefix[i]);
            ans=max(ans,area);
        }
        return ans;
    }
};
