/**
* Previous Greater Element
* Link : https://www.geeksforgeeks.org/problems/previous-greater-element/1
* Topic : Stack & Queue
*/
class Solution {
  public:
    vector<int> preGreaterEle(vector<int>& arr) {
        //  code here
        int n=arr.size();
        stack<int>st;
        vector<int>pge(n);
        
        for(int i=0;i<n;i++) {
            while(!st.empty() && st.top()<=arr[i]) {
                st.pop();
            }
            
            if(st.empty())
                pge[i]=-1;
            else
                pge[i]=st.top();
            
            st.push(arr[i]);
        }
        return pge;
        
    }
};
