/**
* Negative weight cycle
* Link : https://www.geeksforgeeks.org/problems/negative-weight-cycle3504/1
* Topic : Graph
*/
class Solution {
  public:
    int isNegativeWeightCycle(int V, vector<vector<int>> edges) {
        // Code here
        vector<int>dist(V,0);
        
        //iterate for v-1 times
        for(int i=0;i<V-1;i++) {
            //iterate over all the edges
            for(auto it:edges) {
                int u=it[0];
                int v=it[1];
                int wt=it[2];
                
                if(dist[u]!=1e8 && dist[u]+wt<dist[v]) {
                    dist[v]=dist[u]+wt;
                }
            }
        }
        //if we do it again and the value changes then it has negative weights
        for(auto it:edges) {
            int u=it[0];
            int v=it[1];
            int wt=it[2];
            
            if(dist[u]!=1e8 && dist[u]+wt<dist[v]) {
                return 1;
            }
        }
        return 0;
    }
};
