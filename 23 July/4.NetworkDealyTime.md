/**
* 743. Network Delay Time
* Link : https://leetcode.com/problems/network-delay-time/
* Topic : Graph
*/

class Solution {
public:
    int networkDelayTime(vector<vector<int>>& times, int n, int k) {
        vector<vector<pair<int,int>>>adj(n);
        for(auto it:times) {
            int u=it[0]-1;
            int v=it[1]-1;
            int wt=it[2];
            
            adj[u].push_back({v,wt});
        }
        priority_queue<pair<int,int>,vector<pair<int,int>>,greater<pair<int,int>>>pq;
        vector<int>dist(n,1e9);

        k=k-1;
        pq.push({0,k});
        dist[k]=0;
        while(!pq.empty()) {
            int wt=pq.top().first;
            int u=pq.top().second;

            pq.pop();
            for(auto v:adj[u]) {
                int childNode=v.first;
                int childCost=v.second;

                if(wt+childCost<dist[childNode]) {
                    pq.push({wt+childCost,childNode});
                    dist[childNode]=wt+childCost;
                }
            }
        }
        int minTime=(*max_element(dist.begin(),dist.end()));
        if(minTime>=1e9)
            return -1;
        return minTime;
    }
};
