/**
* Missing Coin Sum
* Link : https://cses.fi/problemset/task/2183
* Topic : Greedy
*/
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<long long> a(n);
    for (int i = 0; i < n; i++)
        cin >> a[i];

    sort(a.begin(), a.end());

    long long reachable = 0;

    for (long long coin : a) {
        //coin cannot be created
        if (coin > reachable + 1) {
            cout << reachable + 1;
            return 0;
        }
        else {
            reachable += coin;
        }
        
    }
    cout << reachable + 1;
}
