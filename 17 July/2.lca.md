/**
* LCA in Binary Tree
* Link : https://www.geeksforgeeks.org/problems/lowest-common-ancestor-in-a-binary-tree/1
* Topic : Trees
*/
/*
Definition for Node
class Node {
public:
    int data;
    Node* left;
    Node* right;

    Node(int val) {
        data = val;
        left = right = nullptr;
    }
};
*/

class Solution {
  public:
    Node* lca(Node* root, int n1, int n2) {
        //  code here
        //base case 
        if(root==NULL || root->data==n1 || root->data==n2) {
            return root;
        }
        Node *left=lca(root->left,n1,n2);
        Node *right=lca(root->right,n1,n2);
        
        if(left==NULL)
            return right;
        if(right==NULL)
            return left;
        return root;
    }
};
