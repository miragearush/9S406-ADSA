/**
* Generate all binary strings
* Link : https://www.geeksforgeeks.org/problems/generate-all-binary-strings/1
* Topic : Recursion
*/
class Solution {
  public:
    void solve(int n,string curr,vector<string>&ans) {
        //base condition
        if(curr.size()==n) {
            ans.push_back(curr);
            return;
        }
        
        //put "0"
        
        solve(n,curr+"0",ans);
        //put"1
        solve(n,curr+"1",ans);
    }
    vector<string> binstr(int n) {
        // code here
        vector<string>ans;
        solve(n,"",ans);
        return ans;
        
    }
};
