/**
* 1922. Count Good Numbers
* Link : https://leetcode.com/problems/count-good-numbers/
* Topic : Recursion 
*/
class Solution {
public:
    const int mod=1e9+7;
    long long expo(long long a,long long b) {
    //base case 
    if(b==0)
        return 1;
    // odd case
    if(b%2==1)
        return (a*expo((a*a)%mod,b/2))%mod;
    //even
    return expo((a*a)%mod,b/2);
}
    int countGoodNumbers(long long n) {
        long long oddIndex=n/2;
        long long evenIndex=(n+1)/2;

        long long ans= (expo(5,evenIndex)*expo(4,oddIndex))%mod;
        return (int)ans;
    }
};
