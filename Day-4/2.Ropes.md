#include <bits/stdc++.h>
using namespace std;
 
bool check(double size,vector<int>&a,int k) {
    int count=0;
    for(int i=0;i<a.size();i++) {
        count+=a[i]/size;
    }
    if(count>=k)
        return true;
    return false;
}
 
int main() 
{
    int n,k;
    cin>>n>>k;
 
    vector<int>a(n);
    for(int i=0;i<n;i++) {
        cin>>a[i];
    }
 
    double low =0.0,high=(*max_element(a.begin(),a.end()));
    double ans=0.0;
    
    double precision=1e-7;
 
    while(high-low>precision) {
        double mid = low+(high-low)/2.0;
 
        if(check(mid,a,k)==true) {
            ans=mid;
            low=mid;
        } else {
            high=mid;
        }
    }
    cout<<fixed<<setprecision(7)<<ans;
    return 0;
}
