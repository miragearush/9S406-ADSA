/**
* k-th Smallest in BST
* Link : https://www.geeksforgeeks.org/problems/find-k-th-smallest-element-in-bst/1
* Topic : BST
*/
/*
Definition for Node
class Node {
  public:
    int data;
    Node* left;
    Node* right;

    Node(int val) {
        data = val;
        left = right = nullptr;
    }
};
*/

class Solution {
  public:
    void inorder(Node *node,int &k,int &ans) {
        //base case
        if(node==NULL || k==0)
            return;
        inorder(node->left,k,ans);
        k--;
        if(k==0) {
            ans=node->data;
            return;
        }
        inorder(node->right,k,ans);
    }
    int kthSmallest(Node *root, int k) {
        // code here
        int ans=-1;
        inorder(root,k,ans);
        return ans;
    }
};
