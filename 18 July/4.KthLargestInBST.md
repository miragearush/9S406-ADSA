/**
* Kth Largest in BST
* Link : https://www.geeksforgeeks.org/problems/kth-largest-element-in-bst/1
* Topic : BST
*/
/* Structure of a Binary Tree Node
class Node {
  public:
    int data;
    Node* left;
    Node* right;

    Node(int val) {
        data = val;
        left = right = nullptr;
    }
};*/

class Solution {
  public:
  void inorder(Node *node,int &k,int &ans) {
        //base case
        if(node==NULL || k==0)
            return;
        inorder(node->right,k,ans);
        k--;
        if(k==0) {
            ans=node->data;
            return;
        }
        inorder(node->left,k,ans);
        
    }
    int kthLargest(Node *root, int k) {
        // code here
        int ans=-1;
        inorder(root,k,ans);
        return ans;
    }
};
